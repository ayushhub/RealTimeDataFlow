3    /** 
4     * A custom Key type corresponding to a rectangle 
5     *  
6     *    B-------------------------C 
7     *    |                         | 
8     *    |                         | 
9     *    |                         | 
10    *    |                         | 
11    *    A-------------------------D 
12    */ 
13   import java.io.DataInput; 
14   import java.io.DataOutput; 
15   import java.io.IOException; 
16    
17   import org.apache.hadoop.io.Writable; 
18    
19   public class Rectangle implements Writable{ 
20        
21       public Point2D a; 
22       public Point2D b; 
23       public Point2D c; 
24       public Point2D d; 
25        
26       public Rectangle(Point2D a,Point2D b,Point2D c,Point2D d) { 
27           this.a = a; 
28           this.b = b; 
29           this.c = c; 
30           this.d = d; 
31           System.out.println("Called constructor with points"+a.toString()+b.toString()+c.toString()+d.toString()); 
32       } 
33        
34       public Rectangle() { 
35           Point2D point = new Point2D(0,0); 
36           this.a = point; 
37           this.b = point; 
38           this.c = point; 
39           this.d = point; 
40           System.out.println("Called empty constructor"); 
41       } 
42    
43       @Override 
44       public void readFields(DataInput in) throws IOException { 
45           int p = 0; 
46           System.out.println("Called read on rectangle"); 
47           while (p < 4) { 
48               float x = in.readFloat(); 
49               System.out.println("Read "+x); 
50               float y = in.readFloat(); 
51               System.out.println("Read "+y); 
52               System.out.println("--------"); 
53               if (p == 0) 
54                   a = new Point2D(x,y); 
55               if (p == 1) 
56                   b = new Point2D(x,y); 
57               if (p == 2) 
58                   c = new Point2D(x,y); 
59               if (p == 3) 
60                   d = new Point2D(x,y); 
61               p++; 
62           } 
63           System.out.println("This rectangle is now "+this.toString()); 
64           System.out.println("---------------------------"); 
65       } 
66    
67       @Override 
68       public void write(DataOutput out) throws IOException { 
69           System.out.println("Called write on Rectangle"); 
70           System.out.println(a.x+" "+a.y); 
71           System.out.println(b.x+" "+b.y); 
72           System.out.println(c.x+" "+c.y); 
73           System.out.println(d.x+" "+d.y); 
74           System.out.println("--------------------"); 
75           out.writeFloat(a.x); 
76           out.writeFloat(a.y); 
77           out.writeFloat(b.x); 
78           out.writeFloat(b.y); 
79           out.writeFloat(c.x); 
80           out.writeFloat(c.y); 
81           out.writeFloat(d.x); 
82           out.writeFloat(d.y); 
83            
84       } 
85    
86        
87       @Override  
88       public String toString() { 
89         String returnFormat = "[ (ax,ay),(bx,by),(cx,cy),(dx,dy) ]"; 
90         returnFormat = returnFormat.replace("ax",a.x+""); 
91         returnFormat = returnFormat.replace("ay",a.y+""); 
92         returnFormat = returnFormat.replace("bx",b.x+""); 
93         returnFormat = returnFormat.replace("by",b.y+""); 
94         returnFormat = returnFormat.replace("cx",c.x+""); 
95         returnFormat = returnFormat.replace("cy",c.y+""); 
96         returnFormat = returnFormat.replace("dx",d.x+""); 
97         returnFormat = returnFormat.replace("dy",d.y+""); 
98         return returnFormat; 
99          
100      }
