205      @Override 
206      protected Instances determineOutputFormat(Instances instances) throws Exception { 
207   
208          if (!initSetup) { 
209              result = new Instances(instances, 0); 
210   
211              Enumeration<Object> enumeration = result.attribute(breedIndex).enumerateValues(); 
212   
213              while (enumeration.hasMoreElements()) { 
214                  String breedValue = (String) enumeration.nextElement(); 
215                  breedValue = breedValue.trim(); 
216                  String newBreedValue = breedValue; 
217                  int ind = breedValue.indexOf("/"); 
218                  if (ind != -1) { 
219                      newBreedValue = breedValue.substring(0, ind); 
220                      /* 
221                      String temp = breedValue.substring(ind); 
222                      int ind2 = temp.indexOf(" "); 
223                      if (ind2 != -1) { 
224                          newBreedValue = newBreedValue +temp.substring(ind2+1); 
225                      }*/ 
226                  } 
227                  ind = newBreedValue.indexOf("Mix"); 
228                  if (ind != -1) { 
229                      newBreedValue = newBreedValue.substring(0, ind); 
230                  } 
231                  newBreedValue = newBreedValue.trim(); 
232   
233                  breedValuesMap.put(breedValue,newBreedValue); 
234              } 
235              for(String value : breedValuesMap.values()) { 
236                  newBreedValues.add(value); 
237              } 
238   
239              enumeration = result.attribute(colorIndex).enumerateValues(); 
240   
241              while (enumeration.hasMoreElements()) { 
242                  String colorValue = (String) enumeration.nextElement(); 
243                  colorValue = colorValue.trim(); 
244                  String newColorValue = colorValue; 
245                  int ind = colorValue.indexOf("/"); 
246                  if (ind != -1) { 
247                      newColorValue = colorValue.substring(0, ind); 
248                  } 
249                  ind = newColorValue.indexOf(" "); 
250                  if (ind != -1) { 
251                      newColorValue = newColorValue.substring(0, ind); 
252                  } 
253                  newColorValue = newColorValue.trim(); 
254   
255                  colorValuesMap.put(colorValue,newColorValue); 
256              } 
257   
258              for(String value : colorValuesMap.values()) { 
259                  newColorValues.add(value); 
260              } 
261   
262   
263              System.out.println("should be here only once "); 
264              System.out.println("number of breed values = "+ breedValuesMap.size()); 
265   
266              int nextAttInd = result.numAttributes(); 
267              if (ageUponoutcomeIndex >= 0) { 
268                  result.replaceAttributeAt(new Attribute("Normalized_age"), ageUponoutcomeIndex); 
269                  List<String> ageCat = new ArrayList<String>(); 
270                  ageCat.add(BABY); 
271                  ageCat.add(YOUNG); 
272                  ageCat.add(ADULT); 
273                  ageCat.add(OLD); 
274                  ageCat.add(TEEN); 
275                  result.insertAttributeAt(new Attribute("Age_Category",ageCat),nextAttInd); 
276                  ageCatIndex = nextAttInd; 
277                  nextAttInd++; 
278                  numNewAttrib++; 
279              } 
280              if (noNameIndex >= 0) { 
281                  List<String> hasNameCat = new ArrayList<String>(); 
282                  hasNameCat.add(YES); 
283                  hasNameCat.add(NO); 
284                  result.insertAttributeAt(new Attribute("Has_Name",hasNameCat),nextAttInd); 
285                  noNameIndex = nextAttInd; 
286                  nextAttInd++; 
287                  numNewAttrib++; 
288              } 
289   
290   
291              if (breedIndex >= 0) result.replaceAttributeAt(new Attribute("Breed",new ArrayList<String>(newBreedValues)),breedIndex); 
292              if (colorIndex >= 0) result.replaceAttributeAt(new Attribute("Color",new ArrayList<String>(newColorValues)),colorIndex); 
293   
294              if (dateTimeIndex >= 0) { 
295                  List<String> seasonCat = new ArrayList<String>(); 
296                  seasonCat.add(WINTER); 
297                  seasonCat.add(SUMMER); 
298                  seasonCat.add(SPRING); 
299                  seasonCat.add(FALL); 
300                  //seasonCat.add(UNKNOWN); 
301                  result.insertAttributeAt(new Attribute("Season",seasonCat),nextAttInd); 
302                  seasonIndex = nextAttInd; 
303                  numNewAttrib++; 
304                  nextAttInd++; 
305              } 
306   
307              if (sexUponOutcomeIndex >=0){ 
308                  List<String> sex = new ArrayList<String>(); 
309                  sex.add(MALE); 
310                  sex.add(FEMALE); 
311                  sex.add(UNKNOWN); 
312                  result.insertAttributeAt(new Attribute("Sex",sex),nextAttInd); 
313                  sexIndex = nextAttInd; 
314                  nextAttInd++; 
315                  numNewAttrib++; 
316              } 
317              initSetup = true; 
318   
319          } 
320          return result; 
321      }

323      @Override 
324      protected Instance process(Instance instance) throws Exception { 
325   
326          //double[] values = new double[instance.numAttributes()+numNewAttrib]; 
327          Instance ins = new DenseInstance(instance.numAttributes()+numNewAttrib); 
328          ins.setDataset(determineOutputFormat(instance.dataset())); 
329          try { 
330              for (int n = 0; n < instance.numAttributes(); n++) { 
331   
332                  if (n == ageUponoutcomeIndex) { 
333                      String currentVal = instance.stringValue(n); 
334                      if (currentVal.equals("?")) currentVal = "797 days"; 
335                      String[] parts = currentVal.split(" "); 
336                      int mul = 1; 
337                      if (parts[1].endsWith("s")) parts[1] = parts[1].substring(0, parts[1].length() - 1); 
338                      if (parts[1].equals("month")) { 
339                          mul = 30; 
340                      } else if (parts[1].equals("year")) { 
341                          mul = 365; 
342                      } else if (parts[1].equals("week")) { 
343                          mul = 7; 
344                      } 
345                      //values[n] = Double.parseDouble(parts[0])*mul; 
346                      double age = Double.parseDouble(parts[0]) * mul; 
347                      String ageCat = ADULT; 
348                      if (age < 60) { 
349                          ageCat = BABY; 
350                      } 
351                      else if (age < 1000) 
352                          ageCat = YOUNG; 
353                      else if (age >= 1000 && age <= 1800) { 
354                          ageCat = TEEN; 
355                      } 
356                      else if (age >1800 && age > 3200.0) 
357                          ageCat = OLD; 
358                      ins.setValue(n, age); 
359                      ins.setValue(ageCatIndex,ageCat); 
360   
361                  } else if (n == nameIndex) { 
362                      String noNameVal = NO; 
363                      if (instance.stringValue(n).trim().equals("?")) 
364                          noNameVal = YES; 
365                      //values[noNameIndex] = noNameVal; 
366                      //values[n] = instance.value(n); 
367                      ins.setValue(noNameIndex, noNameVal); 
368                      ins.setValue(n, instance.value(n)); 
369                  } else if (n == breedIndex) { 
370                      String breed = instance.stringValue(n).trim(); 
371                      ins.setValue(n, breedValuesMap.get(breed)); 
372   
373                  } else if (n == colorIndex) { 
374                      String color = instance.stringValue(n).trim(); 
375                      ins.setValue(n, colorValuesMap.get(color)); 
376                  } else if (n == sexUponOutcomeIndex) { 
377                      String sexUponOutcome = instance.stringValue(n).trim(); 
378                      String sex = UNKNOWN; 
379                      if (sexUponOutcome.contains("Male")) 
380                          sex = MALE; 
381                      else if (sexUponOutcome.contains("Female")) 
382                          sex = FEMALE; 
383                      ins.setValue(sexIndex,sex); 
384                      ins.setValue(n,instance.value(n)); 
385                  }else if (n == dateTimeIndex) { 
386                      String dateTime = instance.stringValue(n); 
387                      String season = getSeason(dateTime); 
388                      ins.setValue(seasonIndex,season); 
389                      ins.setValue(n,instance.value(n)); 
390                  } else{ 
391                      //values[n] = instance.value(n); 
392                      ins.setValue(n, instance.value(n)); 
393                  } 
394              } 
395          }catch(Exception e) { 
396             //System.out.println(e.toString()); 
397              e.printStackTrace(); 
398          } 
399          //Instance result = new DenseInstance(1, values); 
400          //System.out.println("The instance: " + ins); 
401          return ins; 
402      }

404      public String getSeason(String datestr) { 
405          SimpleDateFormat sdf = new SimpleDateFormat(DATE_FORMAT); 
406          try { 
407              Date d = sdf.parse(datestr); 
408              Calendar c = Calendar.getInstance(); 
409              c.setTime(d); 
410              int month = c.get(Calendar.MONTH); 
411              month++; 
412              if (month >= 12 || month<=2) 
413                  return WINTER; 
414              if (month >=3 && month <=5) 
415                  return SPRING; 
416              if (month >= 6 && month <=8) 
417                  return SUMMER; 
418              if (month >=9 && month <= 11) 
419                  return FALL; 
420   
421          } catch (ParseException e) { 
422              e.printStackTrace(); 
423          } 
424          return UNKNOWN; 
425      }
