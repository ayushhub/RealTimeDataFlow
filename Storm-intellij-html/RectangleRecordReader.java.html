1    package com.vishnuviswanath.examples.customtypes;

3    import java.io.IOException; 
4     
5    import org.apache.hadoop.io.LongWritable; 
6    import org.apache.hadoop.io.Text; 
7    import org.apache.hadoop.mapred.FileSplit; 
8    import org.apache.hadoop.mapred.JobConf; 
9    import org.apache.hadoop.mapred.LineRecordReader; 
10   import org.apache.hadoop.mapred.RecordReader; 
11        
12   /** 
13    * RecordReader for custom input format RectangleInputFormat 
14    * @author vishnu 
15    * 
16    */ 
17   public class RectangleRecordReader implements RecordReader<Text, RectangleKey> { 
18    
19       private LineRecordReader lineReader; 
20       private LongWritable lineKey; 
21       private Text lineValue; 
22    
23       public RectangleRecordReader(JobConf job,FileSplit split) throws IOException { 
24           lineReader = new LineRecordReader(job,split); 
25           lineKey = lineReader.createKey(); 
26           lineValue = lineReader.createValue(); 
27       } 
28            
29       @Override 
30       public void close() throws IOException { 
31           lineReader.close(); 
32    
33       } 
34    
35       @Override 
36       public Text createKey() { 
37           return new Text(""); 
38       } 
39    
40       @Override 
41       public RectangleKey createValue() { 
42           return new RectangleKey(); 
43       } 
44    
45       @Override 
46       public long getPos() throws IOException { 
47           return lineReader.getPos(); 
48       } 
49    
50       @Override 
51       public float getProgress() throws IOException { 
52           return lineReader.getProgress(); 
53       } 
54    
55       @Override 
56       public boolean next(Text key, RectangleKey value) throws IOException { 
57           //go to the next line for this input split 
58           if (!lineReader.next(lineKey, lineValue)) { 
59               return false; 
60           } 
61    
62           String[] lineParts = lineValue.toString().split(","); 
63           if(lineParts.length != 5) { 
64               System.out.println("Invalid input.\nValid format :[ id,ax ay,bx by,cx cy,dx dy ]"); 
65               return false; 
66           } 
67           float ax,ay,bx,by,cx,cy,dx,dy; 
68           try { 
69               String[] pointA =lineParts[1].split(" "); 
70               String[] pointB =lineParts[2].split(" "); 
71               String[] pointC =lineParts[3].split(" "); 
72               String[] pointD =lineParts[4].split(" "); 
73                
74               ax = Float.parseFloat(pointA[0].trim()); 
75               ay = Float.parseFloat(pointA[1].trim()); 
76               bx = Float.parseFloat(pointB[0].trim()); 
77               by = Float.parseFloat(pointB[1].trim()); 
78               cx = Float.parseFloat(pointC[0].trim()); 
79               cy = Float.parseFloat(pointC[1].trim()); 
80               dx = Float.parseFloat(pointD[0].trim()); 
81               dy = Float.parseFloat(pointD[1].trim()); 
82                
83           } catch (NumberFormatException nfe) { 
84               throw new IOException("Error parsing floating point "+nfe); 
85           } 
86            
87           //set the key as the first part of the record 
88           key.set(lineParts[0].trim()); 
89           //set the value 
90           Point2D a = new Point2D(ax,ay); 
91           Point2D b = new Point2D(bx,by); 
92           Point2D c = new Point2D(cx,cy); 
93           Point2D d = new Point2D(dx,dy); 
94           value.a = a; 
95           value.b = b; 
96           value.c = c; 
97           value.d = d; 
98            
99           return true; 
100      } 
101  }
